public static void
  calculateFormulaResultWithShortestDistanceFromTwoNodes(
GeneralTree tree,
String userInputRawQueries,
boolean printConsoleData) {
ArrayList<int[]> unorderedPairQuery =
      createUnorderedPairsOfIntNumbersFromString
      (userInputRawQueries);
// program time calculation and test cases variables
long totaltime = 0;
int numOfTotalPairs = unorderedPairQuery.size();
long totalResult = 0;
String consoleCalculationProcessMessage = "";
// using a for-each loop, since every pair inside needs to
//undergo the query process
for (int[] query : unorderedPairQuery) {
long startTime = System.currentTimeMillis();
int distance = tree.shortestDistance(query[0],
        query[1]);
// execute formula from the problem
long expressionResult = ( query[0] * query[1] *
        distance ) % 1000000007;
// for the final details of the process, must total
  all the expression result from every query made
totalResult += expressionResult;
// concatenate output message
// if printConsoleData is set to false, then the details
  regarding the process for calculation won't be printed
  to console
if (printConsoleData) {
// create content to print to console, concatenate
  the string with details about the current process
consoleCalculationProcessMessage +=
"Shortest Distance of : " + query[0] + " and " +
           query[1] +
" is " + distance + " : Expression Result ( (" +
            query[0] + " * " + query[1] + " * " + distance
+ " ) % 1,000,000,007( or 10^9 + 7) ) = " +
            expressionResult + "\n";
}
long endTime = System.currentTimeMillis();
totaltime += endTime - startTime;
